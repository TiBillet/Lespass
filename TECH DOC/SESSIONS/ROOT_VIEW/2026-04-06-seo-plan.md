```python
class SEOCache(models.Model):
    """
    Cache SEO pre-calcule par le Celery task refresh_seo_cache.
    Vit dans le schema public (SHARED_APPS). Tous les tenants y accedent en lecture.
    / Pre-computed SEO cache populated by the refresh_seo_cache Celery task.
    Lives in the public schema (SHARED_APPS). All tenants read from it.

    LOCALISATION: seo/models.py
    """

    TENANT_SUMMARY = 'tenant_summary'
    TENANT_EVENTS = 'tenant_events'
    TENANT_MEMBERSHIPS = 'tenant_memberships'
    AGGREGATE_EVENTS = 'aggregate_events'
    AGGREGATE_MEMBERSHIPS = 'aggregate_memberships'
    AGGREGATE_LIEUX = 'aggregate_lieux'
    SITEMAP_INDEX = 'sitemap_index'

    CACHE_TYPE_CHOICES = [
        (TENANT_SUMMARY, _('Tenant summary (config, stats, domain)')),
        (TENANT_EVENTS, _('Published events for tenant')),
        (TENANT_MEMBERSHIPS, _('Published memberships for tenant')),
        (AGGREGATE_EVENTS, _('Aggregated events for network (ROOT)')),
        (AGGREGATE_MEMBERSHIPS, _('Aggregated memberships for network (ROOT)')),
        (AGGREGATE_LIEUX, _('Aggregated active venues (ROOT)')),
        (SITEMAP_INDEX, _('Cross-tenant sitemap index (ROOT)')),
    ]

    cache_type = models.CharField(
        max_length=30,
        choices=CACHE_TYPE_CHOICES,
        db_index=True,
        verbose_name=_("Cache type"),
    )
    tenant = models.ForeignKey(
        'Customers.Client',
        null=True,
        blank=True,
        on_delete=models.CASCADE,
        verbose_name=_("Tenant"),
        help_text=_("null for global aggregates (ROOT)"),
    )
    data = models.JSONField(
        default=dict,
        verbose_name=_("Cached data"),
    )
    updated_at = models.DateTimeField(
        auto_now=True,
        verbose_name=_("Last updated"),
    )

    class Meta:
        unique_together = [('cache_type', 'tenant')]
        verbose_name = _("SEO Cache")
        verbose_name_plural = _("SEO Cache entries")

    def __str__(self):
        tenant_name = self.tenant.name if self.tenant else "global"
        return f"{self.cache_type} — {tenant_name}"
```

```python
import logging
from django.db import connection
from django.core.cache import cache
from Customers.models import Client, Domain

logger = logging.getLogger(__name__)

# TTL du cache Memcached L1 : 4 heures / Memcached L1 cache TTL: 4 hours
SEO_CACHE_TTL = 4 * 60 * 60


def get_active_tenants_with_counts():
    """
    Requete SQL cross-schema unique pour identifier les tenants actifs.
    Retourne les tenants qui ont au moins 1 event OU 1 adhesion publiee.
    / Single cross-schema SQL query to identify active tenants.
    Returns tenants with at least 1 event OR 1 published membership.

    LOCALISATION: seo/services.py
    """
    clients = Client.objects.exclude(
        categorie__in=[Client.ROOT, Client.WAITING_CONFIG]
    ).values_list('uuid', 'schema_name')

    if not clients:
        return []

    # Construction du UNION ALL cross-schema
    # / Build cross-schema UNION ALL
    parts = []
    for client_uuid, schema_name in clients:
        parts.append(f"""
            SELECT
                '{client_uuid}'::uuid AS tenant_id,
                (SELECT COUNT(*) FROM "{schema_name}"."BaseBillet_event"
                 WHERE published = true AND datetime >= NOW()) AS event_count,
                (SELECT COUNT(*) FROM "{schema_name}"."BaseBillet_product"
                 WHERE publish = true AND categorie_article = 'A') AS membership_count
        """)

    if not parts:
        return []

    sql = " UNION ALL ".join(parts)

    with connection.cursor() as cursor:
        cursor.execute(sql)
        rows = cursor.fetchall()

    return [
        {
            'tenant_id': row[0],
            'event_count': row[1],
            'membership_count': row[2],
        }
        for row in rows
        if row[1] > 0 or row[2] > 0
    ]


def get_events_for_tenants(tenant_schemas):
    """
    Recupere les events publies de tous les tenants actifs en une seule requete.
    tenant_schemas : list de (uuid, schema_name)
    / Fetch published events from all active tenants in one query.
    tenant_schemas: list of (uuid, schema_name)

    LOCALISATION: seo/services.py
    """
    if not tenant_schemas:
        return []

    parts = []
    for client_uuid, schema_name in tenant_schemas:
        parts.append(f"""
            SELECT
                '{client_uuid}'::uuid AS tenant_id,
                e.name,
                e.slug,
                e.datetime,
                e.end_datetime,
                e.short_description
            FROM "{schema_name}"."BaseBillet_event" e
            WHERE e.published = true AND e.datetime >= NOW()
        """)

    sql = " UNION ALL ".join(parts) + " ORDER BY datetime ASC"

    with connection.cursor() as cursor:
        cursor.execute(sql)
        columns = [col[0] for col in cursor.description]
        rows = [dict(zip(columns, row)) for row in cursor.fetchall()]

    return rows


def get_memberships_for_tenants(tenant_schemas):
    """
    Recupere les adhesions publiees de tous les tenants actifs en une seule requete.
    / Fetch published memberships from all active tenants in one query.

    LOCALISATION: seo/services.py
    """
    if not tenant_schemas:
        return []

    parts = []
    for client_uuid, schema_name in tenant_schemas:
        parts.append(f"""
            SELECT
                '{client_uuid}'::uuid AS tenant_id,
                p.uuid,
                p.name,
                p.short_description
            FROM "{schema_name}"."BaseBillet_product" p
            WHERE p.publish = true AND p.categorie_article = 'A'
        """)

    sql = " UNION ALL ".join(parts) + " ORDER BY name ASC"

    with connection.cursor() as cursor:
        cursor.execute(sql)
        columns = [col[0] for col in cursor.description]
        rows = [dict(zip(columns, row)) for row in cursor.fetchall()]

    return rows


def build_tenant_config_data(client):
    """
    Switch sur le schema du tenant et recupere les donnees de Configuration.
    Retourne un dict avec les infos SEO du tenant.
    / Switch to tenant schema and fetch Configuration data.
    Returns a dict with tenant SEO info.

    LOCALISATION: seo/services.py
    """
    from django_tenants.utils import tenant_context
    from BaseBillet.models import Configuration

    domain_obj = client.get_primary_domain()
    if not domain_obj:
        return None

    domain = domain_obj.domain

    with tenant_context(client):
        try:
            config = Configuration.get_solo()
        except Exception as exc:
            logger.warning("SEO: impossible de lire Configuration pour %s: %s", client.name, exc)
            return None

        postal = config.postal_address
        city = ""
        country = ""
        if postal:
            city = postal.address_locality or ""
            country = postal.address_country or ""

        social_card_url = config.get_social_card or ""
        logo_url = ""
        if config.logo:
            try:
                logo_url = config.logo.med.url
            except Exception:
                pass

        return {
            "name": config.organisation or client.name,
            "domain": domain,
            "logo_url": logo_url,
            "social_card_url": social_card_url,
            "short_description": config.short_description or "",
            "long_description": config.long_description or "",
            "language": config.language or "fr",
            "city": city,
            "country": country,
            "email": config.email or "",
            "site_web": config.site_web or "",
            "twitter": config.twitter or "",
            "facebook": config.facebook or "",
            "instagram": config.instagram or "",
        }


def set_memcached_l1(cache_type, tenant_uuid, data):
    """
    Ecrit dans le cache Memcached L1 avec le bon TTL.
    / Write to Memcached L1 cache with correct TTL.
    """
    if tenant_uuid:
        key = f"seo:{cache_type}:{tenant_uuid}"
    else:
        key = f"seo:{cache_type}:global"
    cache.set(key, data, SEO_CACHE_TTL)


def get_memcached_l1(cache_type, tenant_uuid=None):
    """
    Lit depuis le cache Memcached L1. Retourne None si miss.
    / Read from Memcached L1 cache. Returns None on miss.
    """
    if tenant_uuid:
        key = f"seo:{cache_type}:{tenant_uuid}"
    else:
        key = f"seo:{cache_type}:global"
    return cache.get(key)
```

```python
import logging
from celery import shared_task
from django.utils import timezone

from Customers.models import Client
from seo.models import SEOCache
from seo.services import (
    get_active_tenants_with_counts,
    get_events_for_tenants,
    get_memberships_for_tenants,
    build_tenant_config_data,
    set_memcached_l1,
)

logger = logging.getLogger(__name__)


@shared_task
def refresh_seo_cache():
    """
    Task Celery qui rafraichit le cache SEO toutes les 4h.
    Idempotent : update_or_create partout.
    / Celery task that refreshes SEO cache every 4h.
    Idempotent: update_or_create everywhere.

    LOCALISATION: seo/tasks.py
    """
    logger.info("SEO: debut du rafraichissement du cache")

    # Etape 1 : tenants actifs via requete SQL cross-schema
    # / Step 1: active tenants via cross-schema SQL query
    active_tenants = get_active_tenants_with_counts()
    active_uuids = {row['tenant_id'] for row in active_tenants}
    counts_by_uuid = {row['tenant_id']: row for row in active_tenants}

    logger.info("SEO: %d tenants actifs trouves", len(active_uuids))

    # Recuperer les objets Client pour les actifs
    # / Fetch Client objects for active tenants
    clients = Client.objects.filter(uuid__in=active_uuids)
    client_by_uuid = {c.uuid: c for c in clients}

    # Construire la liste (uuid, schema_name) pour les requetes SQL
    # / Build (uuid, schema_name) list for SQL queries
    tenant_schemas = [
        (c.uuid, c.schema_name) for c in clients
    ]

    # Etape 2 : events et adhesions en 2 requetes SQL cross-schema
    # / Step 2: events and memberships in 2 cross-schema SQL queries
    all_events = get_events_for_tenants(tenant_schemas)
    all_memberships = get_memberships_for_tenants(tenant_schemas)

    # Regrouper par tenant / Group by tenant
    events_by_tenant = {}
    for ev in all_events:
        tid = ev['tenant_id']
        events_by_tenant.setdefault(tid, []).append(ev)

    memberships_by_tenant = {}
    for mb in all_memberships:
        tid = mb['tenant_id']
        memberships_by_tenant.setdefault(tid, []).append(mb)

    # Etape 3 : Configuration par tenant actif (ORM, set_tenant)
    # / Step 3: Configuration per active tenant (ORM, set_tenant)
    configs_by_uuid = {}
    for client_uuid in active_uuids:
        client = client_by_uuid.get(client_uuid)
        if not client:
            continue
        try:
            config_data = build_tenant_config_data(client)
            if config_data:
                configs_by_uuid[client_uuid] = config_data
        except Exception as exc:
            logger.warning("SEO: erreur config tenant %s: %s", client.name, exc)

    # Etape 4 : ecriture SEOCache + Memcached L1
    # / Step 4: write SEOCache + Memcached L1
    aggregate_events = []
    aggregate_memberships = []
    aggregate_lieux = []
    sitemap_tenants = []

    for client_uuid in active_uuids:
        config = configs_by_uuid.get(client_uuid, {})
        counts = counts_by_uuid.get(client_uuid, {})
        domain = config.get("domain", "")
        client = client_by_uuid.get(client_uuid)

        if not config or not domain:
            continue

        # tenant_summary
        summary_data = {
            **config,
            "event_count": counts.get("event_count", 0),
            "membership_count": counts.get("membership_count", 0),
            "has_upcoming_events": counts.get("event_count", 0) > 0,
        }
        SEOCache.objects.update_or_create(
            cache_type=SEOCache.TENANT_SUMMARY, tenant=client,
            defaults={"data": summary_data},
        )
        set_memcached_l1(SEOCache.TENANT_SUMMARY, client_uuid, summary_data)

        # tenant_events
        tenant_events = []
        for ev in events_by_tenant.get(client_uuid, []):
            event_data = {
                "name": ev["name"],
                "slug": ev["slug"],
                "datetime": ev["datetime"].isoformat() if ev["datetime"] else None,
                "end_datetime": ev["end_datetime"].isoformat() if ev.get("end_datetime") else None,
                "short_description": ev.get("short_description") or "",
                "location": f"{config.get('name', '')}, {config.get('city', '')}",
            }
            tenant_events.append(event_data)
            # Ajouter aux agregats avec infos tenant
            # / Add to aggregates with tenant info
            aggregate_events.append({
                **event_data,
                "tenant_domain": domain,
                "tenant_name": config.get("name", ""),
                "canonical_url": f"https://{domain}/event/{ev['slug']}/",
            })

        events_payload = {"events": tenant_events}
        SEOCache.objects.update_or_create(
            cache_type=SEOCache.TENANT_EVENTS, tenant=client,
            defaults={"data": events_payload},
        )
        set_memcached_l1(SEOCache.TENANT_EVENTS, client_uuid, events_payload)

        # tenant_memberships
        tenant_memberships = []
        for mb in memberships_by_tenant.get(client_uuid, []):
            mb_data = {
                "uuid": str(mb["uuid"]),
                "name": mb["name"],
                "short_description": mb.get("short_description") or "",
            }
            tenant_memberships.append(mb_data)
            aggregate_memberships.append({
                **mb_data,
                "tenant_domain": domain,
                "tenant_name": config.get("name", ""),
                "canonical_url": f"https://{domain}/memberships/{mb['uuid']}/",
            })

        memberships_payload = {"memberships": tenant_memberships}
        SEOCache.objects.update_or_create(
            cache_type=SEOCache.TENANT_MEMBERSHIPS, tenant=client,
            defaults={"data": memberships_payload},
        )
        set_memcached_l1(SEOCache.TENANT_MEMBERSHIPS, client_uuid, memberships_payload)

        # Donnees pour aggregate_lieux
        # / Data for aggregate_lieux
        aggregate_lieux.append({
            "name": config.get("name", ""),
            "domain": domain,
            "logo_url": config.get("logo_url", ""),
            "short_description": config.get("short_description", ""),
```

```python
import json

from seo.models import SEOCache
from seo.services import set_memcached_l1, get_memcached_l1


def get_seo_cache(cache_type, tenant_uuid=None):
    """
    Lecture du cache SEO avec fallback L1 → L2.
    L1 = Memcached, L2 = PostgreSQL (SEOCache).
    Si L1 miss, lit L2 et recharge L1.
    / Read SEO cache with L1 → L2 fallback.
    L1 = Memcached, L2 = PostgreSQL (SEOCache).
    On L1 miss, reads L2 and reloads L1.

    LOCALISATION: seo/views_common.py
    """
    # Essayer L1 (Memcached) / Try L1 (Memcached)
    data = get_memcached_l1(cache_type, tenant_uuid)
    if data is not None:
        return data

    # Fallback L2 (DB) / Fallback to L2 (DB)
    try:
        if tenant_uuid:
            entry = SEOCache.objects.get(cache_type=cache_type, tenant__uuid=tenant_uuid)
        else:
            entry = SEOCache.objects.get(cache_type=cache_type, tenant=None)
        data = entry.data
        # Recharger L1 / Reload L1
        set_memcached_l1(cache_type, tenant_uuid, data)
        return data
    except SEOCache.DoesNotExist:
        return None


def build_json_ld_organization(name, url, logo_url="", description="",
                                same_as=None, address=None):
    """
    Construit un dict JSON-LD schema.org/Organization.
    / Build a JSON-LD schema.org/Organization dict.

    LOCALISATION: seo/views_common.py
    """
    org = {
        "@context": "https://schema.org",
        "@type": "Organization",
        "name": name,
        "url": url,
    }
    if logo_url:
        org["logo"] = logo_url
    if description:
        org["description"] = description
    if same_as:
        org["sameAs"] = same_as
    if address:
        org["address"] = {
            "@type": "PostalAddress",
            **address,
        }
    return org


def build_json_ld_product(name, description, price, currency="EUR", url="",
                           availability="https://schema.org/InStock"):
    """
    Construit un dict JSON-LD schema.org/Product.
    / Build a JSON-LD schema.org/Product dict.

    LOCALISATION: seo/views_common.py
    """
    product = {
        "@context": "https://schema.org",
        "@type": "Product",
        "name": name,
        "description": description,
        "offers": {
            "@type": "Offer",
            "price": str(price),
            "priceCurrency": currency,
            "availability": availability,
        },
    }
    if url:
        product["offers"]["url"] = url
    return product


def build_json_ld_item_list(items, item_type="ListItem"):
    """
    Construit un dict JSON-LD schema.org/ItemList.
    items : liste de dicts avec au minimum "url" et "name".
    / Build a JSON-LD schema.org/ItemList dict.
    items: list of dicts with at least "url" and "name".

    LOCALISATION: seo/views_common.py
    """
    list_items = []
    for position, item in enumerate(items, start=1):
        list_items.append({
            "@type": "ListItem",
            "position": position,
            "url": item.get("url", ""),
            "name": item.get("name", ""),
        })

    return {
        "@context": "https://schema.org",
        "@type": "ItemList",
        "numberOfItems": len(items),
        "itemListElement": list_items,
    }


def robots_txt(request):
```
